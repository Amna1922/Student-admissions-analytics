-- Match by email (most reliable)
UPDATE applicant app
SET sevis_id = sev.sevis_id
FROM sevis sev
WHERE app.official_university_email_address = sev.email
   OR app.email_id = sev.email;

-- Match by name + DOB + university
UPDATE applicant app
SET sevis_id = sev.sevis_id
FROM sevis sev
WHERE LOWER(app.given_name) = LOWER(sev.given_name)
  AND LOWER(app.last_name) = LOWER(sev.surname_or_primary_name)
  AND app.date_of_birth = sev.date_of_birth
  AND app.university ILIKE '%' || split_part(sev.education_level, ' ', 1) || '%'
  AND app.sevis_id IS NULL;

-- See matched records
SELECT COUNT(*) as total_applicants,
       COUNT(sevis_id) as matched_with_sevis,
       COUNT(*) - COUNT(sevis_id) as unmatched
FROM applicant;


-- CONNECT (Strong FK to APPLICANT)
CREATE TABLE deposit (
    id SERIAL PRIMARY KEY,  -- Surrogate primary key
    reference_id BIGINT ,
    deposit_status TEXT,
    assigned TEXT,
    received_at TIMESTAMP,
    university TEXT,
    i_20_status TEXT,
    created_at TIMESTAMP,
    received_date DATE,
    
    CONSTRAINT fk_deposit_applicant
        FOREIGN KEY (reference_id)
        REFERENCES applicant(reference_id)
        ON DELETE CASCADE
    -- NO unique constraint on reference_id - allows duplicates
);

CREATE VIEW applicant_outreach_view AS
SELECT
    a.reference_id,
    a.given_name,
    a.last_name,
    a.email_id,
    o.program_of_interest,
    o.student_status,
    o.appointment_date
FROM applicant a
LEFT JOIN outreach o
  ON LOWER(a.email_id) = LOWER(o.person_email);


-- Insights That Require Further Validation in Week 3

--   Validate SEVIS ID availability by admission decision 
SELECT
    most_recent_released_decision,
    COUNT(*) AS total_applicants,
    COUNT(sevis_id) AS sevis_available,
    COUNT(*) - COUNT(sevis_id) AS sevis_missing
FROM applicant
GROUP BY most_recent_released_decision
ORDER BY total_applicants DESC;


--  Validate outcome quality by application source
SELECT
    a.application_source,
    COUNT(DISTINCT a.reference_id) AS total_applicants,
    COUNT(DISTINCT CASE 
        WHEN a.most_recent_released_decision IS NOT NULL THEN a.reference_id 
    END) AS accepted_applicants,
    COUNT(DISTINCT CASE 
        WHEN d.deposit_status = 'Yes' THEN a.reference_id 
    END) AS deposits_paid
FROM applicant a
LEFT JOIN deposit d
    ON a.reference_id = d.reference_id
GROUP BY a.application_source
ORDER BY total_applicants DESC;

--  Validate visa outcomes by citizenship
-- 
SELECT
    o.citizenship_primary,
    o.student_status,
    COUNT(*) AS student_count
FROM outreach o
WHERE o.student_status IS NOT NULL
GROUP BY o.citizenship_primary, o.student_status
ORDER BY student_count DESC;

-- Validate pipeline progression by program

SELECT
    o.program_of_interest,
    COUNT(*) AS total_students,
    COUNT(CASE WHEN o.accepted_admission = 'Yes' THEN 1 END) AS accepted,
    COUNT(CASE WHEN o.deposit_paid = 'Yes' THEN 1 END) AS deposits_paid,
    COUNT(CASE WHEN o.student_status = 'Visa approved' THEN 1 END) AS visa_approved
FROM outreach o
GROUP BY o.program_of_interest
ORDER BY total_students DESC;


--  Validate intake-based applicant distribution
SELECT
    intake,
    COUNT(*) AS applicant_count
FROM applicant
GROUP BY intake
ORDER BY applicant_count DESC;
